const express = require("express"); const cors = require("cors"); const app = express(); app.use(cors()); app.use(express.json()); app.get("/", (req, res) => { res.json({ name: "Custom Fields Power-Up", capabilities: ["card-buttons", "attachment-sections"], }); }); app.get("/trello-button", (req, res) => { res.json({ text: "Import Data", callback: function (t) { return t.popup({ title: "Upload JSON/CSV", url: "https://example.com/upload", // Change this to your hosting URL }); }, }); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Power-Up running on port ${PORT}`); });
